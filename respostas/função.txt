    function calcularDivisores(A){
        let divisores = []
        let k = 0
        for(let i = 1; i <= A; i++){
            if(A % i == 0){
                divisores[k] = i
                k++
            }
        }
        return divisores 
    }
    let valor = +prompt("insira um valor")
    console.log(calcularDivisores(valor))
