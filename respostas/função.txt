    function organizador(vetor){
        let organizado = []
        let maior = -1
        let posicao 
        for(let j = 49; j >= 0; j--){
            for(let i = 0; i < 50; i++){
                if(Number(vetor[i]) > maior){ 
                    maior = vetor[i]
                    posicao = i
                }
            }
            organizado[j] = maior 
            vetor[posicao] = -1
            maior = -1
        }
        return organizado
    }
    let ram = []
    for(let i = 0; i < 50; i++){
        ram[i] = ((Math.random()) * 1000).toFixed()
    }
    console.log(organizador(ram))

    function calcularBarkara(A, B, C){
        let delta = (B ** 2) - (4 * A * C)
        if(delta < 0) return "não foi posível calcular"
        let resposta1 = (0 - B) + Math.sqrt(delta) / 2 * A
        let resposta2 = (0 - B) - Math.sqrt(delta) / 2 * A
        return `resposta': ${resposta1}\nresposta'': ${resposta2}`
    }
    let A = +prompt("insira o valor de A", 1)
    let B = +prompt("insira o valor de B", 12)
    let C = +prompt("insira o valor de C", -13)
    console.log(calcularBarkara(A, B, C))
